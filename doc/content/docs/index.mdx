## Quick Example

```rust
use gpui::*;
use gpui_component::{button::*, *};

pub struct MyApp;

impl Render for MyApp {
    fn render(&mut self, _: &mut Window, _: &mut Context<Self>) -> impl IntoElement {
        div()
            .v_flex()
            .gap_2()
            .size_full()
            .items_center()
            .justify_center()
            .child("Hello, GPUI Component!")
            .child(
                Button::new("click-me")
                    .primary()
                    .label("Get Started")
                    .on_click(|_, _, _| println!("Clicked!")),
            )
    }
}
```
